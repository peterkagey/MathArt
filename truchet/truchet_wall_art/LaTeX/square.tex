\documentclass[tikz]{standalone}
\usepackage{tikz}
\pgfmathsetmacro{\n}{25}

\begin{document}
  \begin{tikzpicture}
    \foreach \x in {0,...,\n} {
      \foreach \y [
        evaluate={\y as \r using random(0,1)},
        evaluate={\y as \p using int(mod(\r+\x+\y,2))},
        evaluate={\y as \c using 2*(\x+\y)}
      ] in {0,...,\n} {
        \ifnum \p=0 {
          \fill[red!\c!blue, rotate around={90*\r:(\x+0.5,\y+0.5)}]
            (\x,\y) -- (\x,\y+0.5) arc (90:0:0.5) -- cycle
            (\x+1,\y+1) -- (\x+0.5,\y+1) arc (180:270:0.5)--cycle
          ;
        } \else {
          \fill[red!\c!blue, rotate around={90*\r:(\x+0.5,\y+0.5)}]
            (\x,\y+1) -- (\x,\y+0.5) arc (90:0:0.5) -- (\x+1,\y) -- (\x+1,\y+0.5) arc (270:180:0.5) -- cycle
          ;
        }
        \fi
      }
    }
  \end{tikzpicture}
\end{document}
