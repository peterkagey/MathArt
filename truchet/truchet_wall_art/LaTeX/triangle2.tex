\documentclass[tikz]{standalone}
\usepackage{tikz}
\pgfmathsetmacro{\n}{15}
\begin{document}
  \begin{tikzpicture}
    \foreach \b [
      evaluate={\amin using int(-\b/2)},
      evaluate={\amax using int(\n-\b/2)},
      evaluate={\c using 100*\b/\n}
    ] in {0,...,\n} {
      \foreach \a [
        evaluate={\x using \a+\b/2},
        evaluate={\y using \b*sqrt(3)/2},
        evaluate={\r using random(0,9)},
        evaluate={\s using random(0,9)},
        evaluate={\c using 100/\n*\b}
      ] in {\amin,...,\amax} {
        % \draw[fill=red, opacity=0.2]
        %   (\x,\y)--(\x+1,\y)--(\x+0.5,{\y+sqrt(3)/2})--cycle
        % ;
        % \draw[fill=blue, opacity=0.2]
        %   (\x+1.5,{\y+sqrt(3)/2})--(\x+1,\y)--(\x+0.5,{\y+sqrt(3)/2})--cycle
        % ;
        \ifnum \r<4 {
          \ifnum \r = 50 {
            \fill[orange!\c!blue]
              ({\x+1/3},\y) arc (180:0:0.1666) -- cycle
              ({\x+1/6},{\y+sqrt(3)/6}) arc (240:420:0.1666) -- cycle
              ({\x+5/6},{\y+sqrt(3)/6}) arc (300:120:0.1666) -- cycle
            ;
          } \else {
            \fill[orange!\c!blue]
              ({\x+1/3},\y) arc (0:60:0.3333) --
              ({\x+1/3},{\y+sqrt(3)/3}) arc (240:300:0.3333) --
              ({\x+5/6},{\y+sqrt(3)/6}) arc (120:180:0.3333) --
              cycle
            ;
          } \fi
        } \else {
          \fill[orange!\c!blue, rotate around={120*\r:(\x+0.5,{\y+sqrt(3)/6})}]
            ({\x+1/3},\y) arc (180:0:0.1666) -- cycle
            ({\x+1/6},{\y+sqrt(3)/6}) arc (240:300:0.666) --
            ({\x+2/3},{\y+sqrt(3)/3}) arc (300:240:0.333) --
            cycle
          ;
        } \fi

        \ifnum \s<4 {
          \ifnum \s = 50 {
            \fill[orange!\c!blue]
              ({\x+5/6},{\y+sqrt(3)/2}) arc (180:360:0.1666) -- cycle
              ({\x+7/6},{\y+sqrt(3)/6}) arc (240:60:0.1666) -- cycle
              ({\x+5/6},{\y+sqrt(3)/6}) arc (300:480:0.1666) -- cycle
            ;
          } \else {
            \fill[orange!\c!blue]
              ({\x+5/6},{\y+sqrt(3)/6}) arc (120:60:0.333) --
              ({\x+4/3},{\y+sqrt(3)/3}) arc (240:180:0.3333) --
              ({\x+5/6},{\y+sqrt(3)/2}) arc (0:-60:0.3333) -- cycle
            ;
          } \fi
        } \else {
          \fill[orange!\c!blue, rotate around={120*\s:(\x+1,{\y+sqrt(3)/3})}]
            ({\x+5/6}, {\y+sqrt(3)/2}) arc (180:360:0.16666) -- cycle
            ({\x+2/3}, {\y+sqrt(3)/3}) arc (120:60:0.6666) --
            ({\x+7/6}, {\y+sqrt(3)/6}) arc (60:120:0.3333) -- cycle
          ;
        } \fi
      }
    }
  \end{tikzpicture}
\end{document}
