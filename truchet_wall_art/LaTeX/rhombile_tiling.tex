\documentclass[tikz]{standalone}
\usepackage{tikz}
\pgfmathsetmacro{\n}{10}
\begin{document}
  \begin{tikzpicture}
    \foreach \b [
      evaluate={\amin using int(-\b/2)},
      evaluate={\amax using int(\n-\b/2)},
      evaluate={\c using 100*\b/\n}
    ] in {1,...,\n} {
      \foreach \a [
        evaluate={\x using \a+\b/2},
        evaluate={\y using \b*sqrt(3)/2},
        evaluate={\r using random(0,1)},
        evaluate={\s using random(0,1)},
        evaluate={\t using random(0,1)},
      ] in {\amin,...,\amax} {
        % \fill (\x,\y) circle (0.05);
        % \draw (\x+1/2,{\y-1/(2*sqrt(3))})--(\x+1/2,{\y+1/(2*sqrt(3))})--(\x,{\y+1/sqrt(3)}) -- (\x-1/2,{\y+1/(2*sqrt(3))}) -- (\x-1/2,{\y-1/(2*sqrt(3))});
        % \draw[blue] (\x,\y)--(\x+1/2,{\y+1/(2*sqrt(3))})--(\x,{\y+1/sqrt(3)}) -- (\x-1/2,{\y+1/(2*sqrt(3))}) -- cycle;
        % \draw[green] (\x,\y) -- (\x-1/2,{\y+1/(2*sqrt(3))}) -- (\x-1/2,{\y-1/(2*sqrt(3))}) -- (\x,{\y-1/sqrt(3)}) -- cycle;
        % \draw[red] (\x,\y) -- (\x,{\y-1/sqrt(3)}) -- (\x+1/2,{\y-1/(2*sqrt(3))}) -- (\x+1/2,{\y+1/(2*sqrt(3))}) -- cycle;
        \ifnum \r=0 {
          \draw[line width=4.5]
            (\x+1/4, {\y+1/(4*sqrt(3))}) arc (210:150:0.28868)
            (\x-1/4, {\y+1/(4*sqrt(3))}) arc (-30:30:0.28868)
          ;
        } \else {
          \draw[line width=4.5]
            (\x+1/4, {\y+1/(4*sqrt(3))}) arc (30:150:0.28868)
            (\x+1/4, {\y+3/(4*sqrt(3))}) arc (330:210:0.28868)
          ;
        } \fi

        \ifnum \s=0 {
          \draw[line width=4.5]
            (\x+1/4, {\y+1/(4*sqrt(3))}) arc (210:270:0.28868)
            (\x, {\y-1/(2*sqrt(3))}) arc (90:30:0.28868)
          ;
        } \else {
          \draw[line width=4.5]
            (\x+1/4, {\y+1/(4*sqrt(3))}) arc (30:-90:0.28868)
            (\x+1/2, \y) arc (90:210:0.28868)
          ;
        } \fi

        \ifnum \t=0 {
          \draw[line width=4.5]
            (\x-1/2, \y) arc (270:330:0.28868)
            (\x, {\y-1/(2*sqrt(3))}) arc (90:150:0.28868)
          ;
        } \else {
          \draw[line width=4.5]
            (\x-1/2, \y) arc (90:-30:0.28868)
            (\x, {\y-1/(2*sqrt(3))}) arc (270:150:0.28868)
          ;
        } \fi
      }
    }
  \end{tikzpicture}
\end{document}
